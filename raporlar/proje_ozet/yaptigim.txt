  async def _is_interactive(self, file_path: str) -> bool:
      """input() veya sonsuz döngü var mı kontrol et."""
      try:
          from pathlib import Path
          import re
          content = Path(file_path).read_text(encoding="utf-8", errors="ignore")
          patterns = [
              r'\binput\s*\(',
              r'\braw_input\s*\(',
              r'while\s+True\s*:',
              r'while\s+1\s*:',
          ]
          return any(re.search(p, content) for p in patterns)
      except Exception:
